# ENCUENTRA LA INTEGRAL
# *********************


def run(symbol: str) -> str:
    # Encontrar la posición de la coma en la cadena
    coma_pos = symbol.find(',')

    # Obtener el coeficiente y el exponente como subcadenas
    coeficiente = int(symbol[:coma_pos])
    exponente = int(symbol[coma_pos + 1 :])

    # Calcular la integral
    nuevo_exponente = exponente + 1
    nuevo_coeficiente = coeficiente // nuevo_exponente

    # Crear la expresión de la integral
    integral = f"{nuevo_coeficiente}x^{nuevo_exponente}"

    return integral


if __name__ == '__main__':
    run('3,2')
